> Complete API reference for @vacano/ui React component library.
> Download: https://ui.vacano.io/MLAPI.md

## Overview

```tsx
// Main components
import { Button, Input, OtpCode, Select, Checkbox, Modal, ... } from '@vacano/ui'

// Icons (1800+ Lucide icons)
import { IconSearch, IconUser, IconPlus, ... } from '@vacano/ui/icons'

// Utilities
import { COLORS, BREAKPOINTS, useKeyBinding } from '@vacano/ui/lib'
```

## Setup

```tsx
import { GlobalStyle } from '@vacano/ui'

function App() {
  return (
    <>
      <GlobalStyle />
      {/* Your app */}
    </>
  )
}
```

# FORM COMPONENTS

## Button

Clickable button with variants, icons, keyboard bindings, and loading state.

```tsx
import { Button } from '@vacano/ui'
import { IconPlus, IconTrash2, IconSearch } from '@vacano/ui/icons'
```

### Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| children | string | - | Button label |
| variant | 'normal' \| 'system' \| 'success' \| 'warning' \| 'danger' \| 'transparent' | 'normal' | Visual style |
| size | 'compact' \| 'default' | 'default' | Button size |
| type | 'button' \| 'submit' \| 'reset' \| 'link' | 'button' | Button type |
| icon | ReactNode | - | Icon element |
| keyBindings | KeyboardEventKey[] | - | Keyboard shortcut |
| loading | boolean | false | Show spinner |
| disabled | boolean | false | Disable button |
| fullWidth | boolean | false | Full width |
| onClick | () => void | - | Click handler |

### Examples

```tsx
// Basic variants
<Button variant="normal">Normal</Button>
<Button variant="system">System</Button>
<Button variant="success">Success</Button>
<Button variant="warning">Warning</Button>
<Button variant="danger">Delete</Button>
<Button variant="transparent">Cancel</Button>

// Sizes
<Button size="default">Default</Button>
<Button size="compact">Compact</Button>

// With icon
<Button icon={<IconPlus />}>Add Item</Button>
<Button icon={<IconTrash2 />} variant="danger">Delete</Button>

// Icon only
<Button icon={<IconSearch />} variant="transparent" />

// Keyboard bindings
<Button keyBindings={['Meta', 'S']}>Save</Button>
<Button keyBindings={['Meta', 'Enter']}>Submit</Button>
<Button keyBindings={['Escape']} variant="transparent">Cancel</Button>

// Loading state
<Button loading>Saving...</Button>
<Button loading variant="system">Processing</Button>

// Full width
<Button fullWidth variant="success">Continue</Button>

// Combined
<Button
  icon={<IconPlus />}
  keyBindings={['Meta', 'N']}
  variant="normal"
>
  New Item
</Button>
```

### Layout Patterns

```tsx
// Button group (row)
<div style={{ display: 'flex', gap: 12 }}>
  <Button variant="system">Cancel</Button>
  <Button variant="normal">Save</Button>
</div>

// Right-aligned actions
<div style={{ display: 'flex', justifyContent: 'flex-end', gap: 12 }}>
  <Button variant="transparent">Cancel</Button>
  <Button variant="danger">Delete</Button>
</div>

// Dialog footer
<div style={{ display: 'flex', justifyContent: 'space-between' }}>
  <Button variant="transparent">Back</Button>
  <div style={{ display: 'flex', gap: 12 }}>
    <Button variant="system">Cancel</Button>
    <Button variant="normal">Continue</Button>
  </div>
</div>

// Toolbar with icons
<div style={{ display: 'flex', gap: 8 }}>
  <Button icon={<IconPlus />} size="compact" />
  <Button icon={<IconEdit />} size="compact" />
  <Button icon={<IconTrash2 />} size="compact" variant="danger" />
</div>
```

## OtpCode

OTP code input for verification codes (SMS, 2FA).

```tsx
import { OtpCode } from '@vacano/ui'
```

### Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| value | string | '' | Current OTP value |
| onChange | (value: string) => void | - | Change handler |
| length | number | 6 | Number of digits |
| label | string | - | Label text |
| variant | 'normal' \| 'error' | 'normal' | Visual style |
| size | 'compact' \| 'default' | 'default' | Input size |
| disabled | boolean | false | Disable input |

### Examples

```tsx
const [code, setCode] = useState('')

// Basic
<OtpCode value={code} onChange={setCode} />

// With label
<OtpCode label="Verification code" value={code} onChange={setCode} />

// Different lengths
<OtpCode length={4} label="4-digit code" value={code} onChange={setCode} />
<OtpCode length={6} label="6-digit code" value={code} onChange={setCode} />
<OtpCode length={8} label="8-digit code" value={code} onChange={setCode} />

// Variants
<OtpCode variant="normal" label="Enter code" value={code} onChange={setCode} />
<OtpCode variant="error" label="Invalid code" value={code} onChange={setCode} />

// Sizes
<OtpCode size="default" value={code} onChange={setCode} />
<OtpCode size="compact" value={code} onChange={setCode} />

// Disabled
<OtpCode disabled value="123456" />
```

### Features

- Auto-focus next input on digit entry
- Arrow key navigation (left/right)
- Backspace deletes and moves focus back
- Paste support (Ctrl+V)
- Numeric input only
- `autoComplete="one-time-code"` for SMS autofill

### Layout Patterns

```tsx
// Centered verification form
<div style={{ display: 'flex', flexDirection: 'column', alignItems: 'center', gap: 16 }}>
  <h2>Verify your identity</h2>
  <p>Enter the 6-digit code sent to your email</p>
  <OtpCode value={code} onChange={setCode} />
  <Button variant="normal">Verify</Button>
</div>

// With error state
const [error, setError] = useState(false)

<OtpCode
  value={code}
  onChange={(v) => { setCode(v); setError(false) }}
  variant={error ? 'error' : 'normal'}
  label={error ? 'Invalid code' : 'Verification code'}
/>
```
